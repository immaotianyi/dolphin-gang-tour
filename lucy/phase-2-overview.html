③ 核心代码 — badusb_guard.rs (安全审查)
→ 设计要点: 三级审查体系。STRING 命令的多行拼接通过 string_buffer 追踪累积，先匹配白名单（跳过危险检查），再匹配 25 项危险模式。ENTER 后检查是否刚打开了终端，防止命令执行。
src/security/badusb_guard.rsRust
// 25 项危险命令拦截
const DANGER_PATTERNS: &[(&str, &str, &str)] = &[
    // 系统破坏 (8)
    ("rm -rf", "D001", "Recursive force delete"),
    ("format ", "D002", "Disk format command"),
    // ... 共 25 项
];

// 审查主函数
pub fn analyze(script: &str) -> GuardReport {
    let mut string_buffer = String::new();  // 追踪 STRING 拼接

    for (i, line) in lines.iter().enumerate() {
        if lower.starts_with("string ") {
            string_buffer.push_str(&text);
            // 1. 先检查白名单
            let is_safe = SAFE_PATTERNS.iter()
                .any(|(pat, _, _)| string_buffer.contains(pat));
            if !is_safe {
                // 2. 匹配 25 项危险模式
                for (pattern, rule_id, desc) in DANGER_PATTERNS {
                    if string_buffer.contains(pattern) {
                        issues.push(GuardIssue {
                            severity: Severity::Danger, ...
                        });
                    }
                }
                // 3. URL + 下载关键词组合检测
                if string_buffer.contains("http") && string_buffer.contains("download") {
                    issues.push(GuardIssue { rule_id: "D026", ... });
                }
            }
        }
        // 4. 检查 ENTER 后的命令执行
        if lower == "enter" && prev.contains("terminal") {
            issues.push(GuardIssue { rule_id: "D027", ... });
        }
    }
    GuardReport { passed: danger_count == 0, ... }
}
→ 安全门控流程: 前端调用 badusb_validate → 返回 GuardReport → 如果 danger_count > 0 → 前端弹窗确认 → 用户确认后 badusb_execute 带 force=true → 设备执行 HID 注入。

// 已知卡片类型数据库: (UID前缀, 类型, 厂商, ATQA, SAK)
const CARD_DATABASE: &[(&str, &str, &str, u16, u8)] = &[
    ("04", "NTAG213",   "NXP", 0x0044, 0x00),
    ("08", "Mifare Classic 1K", "NXP", 0x0004, 0x08),
    ("20", "Mifare DESFire", "NXP", 0x0344, 0x20),
    // ... 共 8 种卡片类型
];

// 通过 UID 前缀 + ATQA/SAK 识别卡片
fn identify_card(uid: &str, atqa: u16, sak: u8) -> (String, String) {
    let uid_prefix = uid.split(':').next().unwrap_or("").to_lowercase();
    for (prefix, card_type, manufacturer, exp_atqa, exp_sak) in CARD_DATABASE {
        if uid_prefix.starts_with(prefix) && (atqa == *exp_atqa || sak == *exp_sak) {
            return (card_type.to_string(), manufacturer.to_string());
        }
    }
    ("Unknown".to_string(), "Unknown".to_string())
}

pub const KNOWN_PROTOCOLS: &[(u32, &str, &str)] = &[
    (433_920_000, "PT2262/EV1527", "Wireless doorbell / remote control"),
    (315_000_000, "Car Keyless", "Vehicle remote entry system"),
    (433_075_000, "KeeLoq", "Rolling code remote (encrypted)"),
    // ... 共 7 种协议
];

// 频率 ±500kHz 容差匹配
pub fn identify_protocol(freq: u32) -> Option<(&static str, &static str)> {
    KNOWN_PROTOCOLS.iter()
        .find(|(f, _, _)| (*f as i64 - freq as i64).abs() < 500_000)
        .map(|(_, name, desc)| (*name, *desc))
}

// 验证频率是否在合法频段内
pub fn is_legal_frequency(freq: u32) -> bool {
    FREQ_BANDS.iter().any(|(start, end, _)| freq >= *start && freq <= *end)
}
