③ 核心代码 — UsbTransport (reader thread 模式)
→ 设计要点: reader thread 持续读串口 → 解析帧 → 有 id 字段则匹配 pending oneshot → 有 evt 字段则 emit 前端事件。send_command 用 oneshot channel 实现请求/响应多路复用。
src/device/usb_transport.rsRust
// 真实 USB 设备传输层 — 核心结构
pub struct UsbTransport {
    // 串口句柄（共享给 reader thread）
    port: Arc<Mutex<Box<dyn serialport::SerialPort>>>,
    // 待响应请求表: transaction_id → oneshot::Sender
    pending: Arc<Mutex<HashMap<u16, oneshot::Sender<serde_json::Value>>>>,
    next_id: AtomicU16,        // 事务 ID 自增
    connected: AtomicBool,     // 连接状态（无锁读）
    reader_handle: Mutex<Option<JoinHandle>>,
}

// send_command — 请求/响应多路复用
async fn send_command(&self, module: &str, op: &str, data: serde_json::Value) -> LucyResult<serde_json::Value> {
    let id = self.alloc_id();
    let frame = protocol::encode_request(id, module, op, &data)?;

    // 1. 注册 pending 请求
    let (tx, rx) = oneshot::channel();
    self.pending.lock().insert(id, tx);

    // 2. 写入串口
    self.port.lock().write_all(&frame)?;

    // 3. 等待响应（5s 超时）
    match tokio::time::timeout(Duration::from_millis(5000), rx).await {
        Ok(Ok(response)) => Ok(response),
        Err(_) => Err(LucyError::Usb("Request timeout".into())),
    }
}

// reader_loop — 后台线程持续读串口并分发
fn reader_loop(port: Arc<...>, pending: Arc<...>, app: AppHandle) {
    loop {
        let n = port.lock().read(&mut buf)?;  // 阻塞读
        frame_buf.extend(&buf[..n]);

        loop {
            match try_parse_frame(&frame_buf[scan_pos..]) {
                Complete(data, consumed) => {
                    let value = protocol::decode_response(&data)?;
                    if let Some(id) = value.get("id") {
                        // 响应帧 → 匹配 pending → oneshot 唤醒
                        pending.lock().remove(&id).send(value);
                    } else if let Some(evt) = value.get("evt") {
                        // 事件帧 → emit 前端
                        app.emit("screen_frame", payload);  // 屏幕帧
                        app.emit("subghz_signal", payload); // SubGHz 信号
                        app.emit("nfc_detected", payload);  // NFC 检测
                    }
                }
                Partial => break,  // 需要更多数据
                Invalid(pos) => scan_pos += pos,  // 跳过垃圾字节
            }
        }
    }
}
→ 关键决策: 使用 std::thread 而非 tokio task 读串口，因为 serialport::SerialPort::read 是阻塞 API。通过 oneshot channel 与 tokio 异步世界桥接。

// 自动连接 — 优先真实设备，降级虚拟设备
pub async fn auto_connect(&self, app: &AppHandle) -> LucyResult<bool> {
    match detector::scan_devices().await {
        Ok(devices) => {
            // 查找 Lucy 设备 (VID=0x303A, PID=0x4001)
            if let Some(dev) = devices.iter().find(|d| d.vid == LUCY_VID && d.pid == LUCY_PID) {
                self.connect(&dev.port, app).await?;
                return Ok(true);  // 真实设备
            }
            // 降级 → VirtualDevice
            self.connect("VIRTUAL", app).await?;
            let _ = app.emit("device_fallback", json!({"reason": "No Lucy device"}));
            Ok(false)  // 虚拟设备
        }
        Err(e) => { // 扫描失败也降级
            self.connect("VIRTUAL", app).await?;
            Ok(false)
        }
    }
}
⑤ 核心代码 — 帧协议 (MessagePack + CRC16)
src/device/protocol.rsRust
// 帧格式: [magic(0xAA55)] [ver(u8)] [len(u16)] [payload(msgpack)] [crc16(u16)]
pub const MAGIC: [u8; 2] = [0xAA, 0x55];
pub const MAX_FRAME_SIZE: usize = 128 * 1024; // 128KB

pub fn encode_request(id: u16, module: &str, op: &str, data: &Value) -> LucyResult<Vec<u8>> {
    let payload = json!({"id": id, "mod": module, "op": op, "data": data});
    let bytes = rmp_serde::to_vec_named(&payload)?;  // MessagePack 编码
    let crc = crc16::State::<crc16::ARC>::calculate(&bytes);

    let mut frame = Vec::with_capacity(7 + bytes.len());
    frame.extend(&MAGIC);                    // 0xAA55
    frame.push(PROTOCOL_VERSION);       // 0x01
    frame.extend(&(bytes.len() as u16).to_le_bytes());  // payload length
    frame.extend(&bytes);                    // MessagePack payload
    frame.extend(&crc.to_le_bytes());        // CRC16
    Ok(frame)
}
→ 为什么 MessagePack: 比 JSON 小 30-50%，二进制安全，ESP32-S3 端有成熟的 msgpack-c 库。CRC16-ARC 检测传输错误，magic bytes 实现帧同步。

pub fn run() {
    let shared_state: SharedState = Arc::new(RwLock::new(AppState::new()));
    let transport_manager = Arc::new(TransportManager::new());  // Arc 包装

    tauri::Builder::default()
        .manage(shared_state)
        .manage(transport_manager)
        .setup(|app| {
            // 后台任务 1: 设备自动扫描（每 3s）
            tauri::async_runtime::spawn(async {
                detector::start_auto_scan(handle).await;
            });

            // 后台任务 2: 自动连接（延迟 2s 等前端加载）
            tauri::async_runtime::spawn(async {
                sleep(Duration::from_secs(2)).await;
                let tm = handle.state::<Arc<TransportManager>>().inner().clone();
                match tm.auto_connect(&handle).await {
                    Ok(real) => {
                        // 更新 SSOT 状态 → emit 前端
                        state.write().connection_state = Connected;
                        emit_state_update(&handle, &state);
                        // 虚拟设备 → 启动屏幕帧推送
                        if !real { start_virtual_screen_stream_public(handle).await; }
                    }
                    Err(e) => { state.write().connection_state = Error; }
                }
            });
            Ok(())
        })
        .invoke_handler(generate_handler![
            device_scan, device_connect, device_disconnect,
            device_get_info, device_refresh_info,
            nfc_detect, nfc_read_uid,
            subghz_scan, subghz_rx,
            gpio_set_direction, gpio_set_value, gpio_read,
            ir_learn, ir_transmit,
            screen_get_frame, ai_send_message,
            get_app_state, close_window, minimize_window,
        ])
        .run(generate_context!())
}
