③ 核心代码 — sanitizer.rs (7 种脱敏)
→ 设计要点: 使用 OnceLock 避免重复编译正则。脱敏后保存 HashMap<placeholder, original>，AI 回复中可还原。只对云端 LLM 脱敏，本地模型不脱敏。
src/ai/sanitizer.rsRust
// 7 种脱敏模式 — OnceLock 避免重复编译
fn patterns() -> &'static [(Regex, &'static str, &'static str)] {
    static PATTERNS: OnceLock<Vec<...>> = OnceLock::new();
    PATTERNS.get_or_init(|| vec![
        // 1. NFC UID — 04:A3:B2:C1
        (Regex::new(r"(?i)\b([0-9a-f]{2}:){3,9}[0-9a-f]{2}\b").unwrap(), "uid", "[UID_REDACTED]"),
        // 2. NFC Key — A1B2C3D4E5F6
        (Regex::new(r"(?i)\b[0-9a-f]{12}\b").unwrap(), "key", "[KEY_REDACTED]"),
        // 3. WiFi — password: xxxxx
        (Regex::new(r"(?i)(?:wifi|password|psk)\s*[:=]\s*(\S{8,63})").unwrap(), "wifi", "[WIFI_REDACTED]"),
        // 4. API Key — sk-xxx / Bearer xxx
        (Regex::new(r"(?i)(?:sk-|bearer |api[_-]?key\s*[:=]\s*)([a-z0-9_\-]{20,})").unwrap(), "apikey", "[API_KEY_REDACTED]"),
        // 5. 中国手机号 — 13812345678
        (Regex::new(r"\b1[3-9]\d{9}\b").unwrap(), "phone", "[PHONE_REDACTED]"),
        // 6. 邮箱 — test@example.com
        (Regex::new(r"\b[\w.%+-]+@[\w.-]+\.[a-zA-Z]{2,}\b").unwrap(), "email", "[EMAIL_REDACTED]"),
        // 7. GPS 坐标 — 31.2304,121.4737
        (Regex::new(r"\b-?\d{1,3}\.\d{4,8}\s*,\s*-?\d{1,3}\.\d{4,8}\b").unwrap(), "coords", "[COORDS_REDACTED]"),
    ])
}

// 脱敏 + 保存映射表 (用于 AI 回复中还原)
pub fn sanitize(input: &str) -> SanitizedText {
    let mut text = input.to_string();
    let mut replacements = HashMap::new();
    for (regex, _, placeholder) in patterns() {
        text = regex.replace_all(&text, |caps: &Captures| {
            let full = caps.get(0).map(|m| m.as_str()).unwrap_or("");
            replacements.insert(placeholder.to_string(), full.to_string());
            placeholder.to_string()
        }).to_string();
    }
    SanitizedText { text, replacements }
}

// 4 个 Provider — 统一 OpenAI 兼容 API
pub enum Provider { Deepseek, Openai, Qwen, Local }

impl Provider {
    pub fn endpoint(&self) -> &str {
        match self {
            Deepseek => "https://api.deepseek.com/v1/chat/completions",
            Openai   => "https://api.openai.com/v1/chat/completions",
            Qwen     => "https://dashscope.aliyuncs.com/compatible-mode/v1/chat/completions",
            Local    => "http://localhost:11434/v1/chat/completions",  // Ollama
        }
    }
    pub fn default_model(&self) -> &str {
        match self {
            Deepseek => "deepseek-chat",
            Openai   => "gpt-4o-mini",
            Qwen     => "qwen-plus",
            Local    => "llama3.2",
        }
    }
    pub fn needs_api_key(&self) -> bool { !matches(self, Local) }
}

// 解析 SSE 流中的一行
pub fn parse_sse_line(line: &str) -> Option<String> {
    if !line.starts_with("data: ") { return None; }
    let data = &line[6..];
    if data.trim() == "[DONE]" { return None; }
    let chunk: StreamChunk = serde_json::from_str(data).ok()?;
    chunk.choices.first()?.delta.content
}
⑤ 核心代码 — pipeline.rs (SSE 流式 + 命令解析)
→ 设计要点: run_pipeline 是核心函数。逐行解析 SSE stream，每个 token 通过 app.emit("ai_token", token) 推送到前端。累积完整回复后解析 <cmds> 标签提取设备命令。无 API Key 时降级为 local_response()。
src/ai/pipeline.rsRust
// AI 管道主函数
pub async fn run_pipeline(
    app: &AppHandle, message: &str, model: &str,
    device_context: &str, history: &[ChatMessage],
) -> LucyResult<AiResponse> {
    let config = provider::load_config(model);

    // 1. 脱敏 (仅云端 LLM)
    let sanitized = if config.provider.needs_api_key()
        && sanitizer::contains_sensitive(message) {
        sanitizer::sanitize(message)
    } else { /* no-op */ };

    // 2. 无 API Key → 本地降级
    if config.provider.needs_api_key() && config.api_key.is_empty() {
        return Ok(local_response(&sanitized.text));
    }

    // 3. 构建消息: system prompt + history + user
    let mut messages = vec![ChatMessage::system(&build_system_prompt(device_context))];
    messages.extend_from_slice(history);
    messages.push(ChatMessage::user(&sanitized.text));

    // 4. 发起 SSE 请求
    let request = provider::build_request(&config, messages, 60)?;
    let response = request.send().await?;

    // 5. 流式读取 SSE → 逐 token emit
    let mut stream = response.bytes_stream();
    let mut full_content = String::new();
    let mut buffer = String::new();

    while let Some(chunk) = stream.next().await {
        let text = String::from_utf8_lossy(&chunk?);
        buffer.push_str(&text);
        while let Some(pos) = buffer.find('\n') {
            let line = buffer[..pos].to_string();
            buffer = buffer[pos + 1..].to_string();
            if let Some(token) = provider::parse_sse_line(&line) {
                full_content.push_str(&token);
                let _ = app.emit("ai_token", &token);  // → 前端逐字渲染
            }
        }
    }

    // 6. 解析 <cmds> 标签
    let cmds = parse_commands(&full_content);

    // 7. 还原脱敏数据
    let content = sanitizer::restore(&full_content, &sanitized.replacements);

    Ok(AiResponse { content, cmds, sanitized: !sanitized.replacements.is_empty(), model })
}

// <cmds> 标签解析
pub fn parse_commands(content: &str) -> Vec<DeviceCommand> {
    for caps in CMDS_REGEX.captures_iter(content) {
        for line in block.lines() {
            let parts: Vec<&str> = line.split_whitespace().collect();
            // module action args...
            DeviceCommand { module: parts[0], action: parts[1], args: parts[2..] }
        }
    }
}

pub fn local_response(message: &str) -> AiResponse {
    let lower = message.to_lowercase();
    let content = if lower.contains("nfc") {
        "I can help you with NFC! Lucy supports reading Mifare Classic, NTAG, and DESFire cards.

<cmds>
nfc detect
</cmds>

Click 'Execute' to scan for a card."
    } else if lower.contains("subghz") {
        "SubGHz module supports 300-928MHz frequency range.

<cmds>
subghz rx 433920000
</cmds>

This will listen on 433.92MHz (common doorbell/remote frequency)."
    } else if lower.contains("gpio") {
        "Lucy has 8 GPIO pins:
- GPIO1-2: ADC inputs
- GPIO3-4: UART
- GPIO5-6: I2C
- GPIO7-8: SPI

<cmds>
gpio scan
</cmds>"
    } // ... ir, badusb, hello, default;

    AiResponse { content, cmds: parse_commands(content), model: "local" }
}
